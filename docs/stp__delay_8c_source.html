45float stp_delay_interpolate_tap_into_buffer(long buffer_size, float delay_in_samples, float *buffer_pointer, float *circular_pointer)
46{
47 // positions in buffer to interpolate between
48 unsigned int i, j;
49 float si, sj;
50
51 // interpolate between si and sj
52 i = floor(delay_in_samples);
53 // if i=D, then j=0 otherwise, j=i+1
54 j = (i + 1) % ((buffer_size) + 1);
55
56 // get buffer values by taping into buffer without interpolation
57 // si(n) = x(n-i)
58 si = stp_delay_tap_into_buffer(buffer_size, buffer_pointer, circular_pointer, i);
59 // sj(n) = x(n-j)
60 sj = stp_delay_tap_into_buffer(buffer_size, buffer_pointer, circular_pointer, j);
61
62 return si + ((delay_in_samples) - i) * (sj - si);
63}
